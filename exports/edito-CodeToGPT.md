class MediaService {
  /// Request permissions dynamically based on Android version
  static Future<bool> requestPermission() async {
    // For Android 13+ (SDK 33)
    if (Platform.isAndroid) {
      final videos = await Permission.videos.request();
      final photos = await Permission.photos.request();

      if (videos.isGranted || photos.isGranted) return true;
      if (videos.isLimited || photos.isLimited) return true;
    }

    // Fallback for older versions / iOS
    final PermissionState ps = await PhotoManager.requestPermissionExtend();
    return ps.isAuth || ps.hasAccess;
  }

  /// Fetch recent videos with pagination support
  static Future<List<AssetEntity>> loadRecentVideos({int page = 0, int size = 80}) async {
    final List<AssetPathEntity> albums = await PhotoManager.getAssetPathList(
      type: RequestType.video,
      filterOption: FilterOptionGroup(
        orders: [const OrderOption(type: OrderOptionType.createDate, asc: false)],
      ),
    );

    if (albums.isEmpty) return [];

    // Get the "Recent" album (usually the first one)
    final AssetPathEntity recentAlbum = albums.first;

    // Fetch assets for the requested page
    return await recentAlbum.getAssetListPaged(page: page, size: size);
  }
}
